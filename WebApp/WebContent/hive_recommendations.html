recommendations.q

use store;

 

insert overwrite table recommendation_summary_hive

select recommendation, count(*)

from product

group by recommendation;

 

 

insert overwrite table hr_products

select p.recommendation, count(*), os.product_id, p.product_name

from product p

join orderstream os

on p.product_id = os.product_id

group by p.recommendation, os.product_id, p.product_name;

 

 

 

insert overwrite table vendor_hr_products

select p.vendor, count(*), os.product_id, p.product_name

from product p

join orderstream os

on p.product_id = os.product_id

where p.recommendation = 'highly recommend'

group by p.vendor, os.product_id, p.product_name;
